Авторизация:
	CLient->Server:
		JSON:
			{
			auth: 
				{
				login: "%login%",
				pass: "%pass%"
				}
			}
		
		%login% - логин
		%pass% - пароль
			
	Server->Client:
		JSON:
			{
			response: "%response%",
			token: "%token%"
			}
			
		%response% - ответ сервера. "Error N" or "OK"
			"Error N" - ошибка N => %token% = ""
			"OK" - ошибки нет
		%token% - идентификатор пользователя, для дальнейших запросов

Список друзей:
	Client->Server:
		JSON:
			{
			friends: 
				{
					token: "%token%"
				}
			}
		%token% - идентификатор пользователя
			
	Server->Client:
		JSON:
			{
			response: %response%,
			friends: [%friendsList%]
			}
			
		%response% - ответ сервера. "Error N" or "OK"
			"Error N" - ошибка N => %token% = ""
			"OK" - ошибки нет
		%friendsList% - список друзей
