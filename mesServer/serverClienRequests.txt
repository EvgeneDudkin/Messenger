Авторизация:
	CLient->Server:
		JSON:
			{
			auth: 
				{
				login: "%login%",
				pass: "%pass%"
				}
			}
		
		%login% - логин
		%pass% - пароль
			
	Server->Client:
		JSON:
			{
			response: "%response%",
			token: "%token%",
			dialogs: [%dialogsList%]
			}
			
		%response% - ответ сервера. "Error N" or "OK"
			"Error N" - ошибка N => %token% = ""
			"OK" - ошибки нет
		%token% - идентификатор пользователя, для дальнейших запросов
		%dialogsList% - список друзей вида {id: %id%, name: %name%}

			}

			{

			}
